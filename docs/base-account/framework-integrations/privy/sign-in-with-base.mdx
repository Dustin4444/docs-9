Learn how to implement Sign in with Base using Privy to authenticate users and connect them with Base Account.

## Prerequisites

Make sure you have [set up Privy with Base Account](/base-account/framework-integrations/privy/setup) before following this guide.

## Authentication Flow

Privy provides multiple authentication options that can be combined with Base Account:

### 1. Basic Authentication Component

Create a comprehensive authentication component:

```tsx
// components/BaseAccountAuth.tsx
import { usePrivy } from '@privy-io/react-auth'
import { useWallets } from '@privy-io/react-auth'
import { useState } from 'react'

export function BaseAccountAuth() {
  const { ready, authenticated, user, login, logout } = usePrivy()
  const { wallets, ready: walletsReady } = useWallets()
  const [isConnecting, setIsConnecting] = useState(false)

  const baseAccountWallet = wallets.find(
    wallet => wallet.walletClientType === 'base_account'
  )

  const handleSignInWithBase = async () => {
    setIsConnecting(true)
    try {
      await login()
    } catch (error) {
      console.error('Login failed:', error)
    } finally {
      setIsConnecting(false)
    }
  }

  if (!ready || !walletsReady) {
    return (
      <div className="flex items-center justify-center p-8">
        <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-blue-500"></div>
      </div>
    )
  }

  if (!authenticated) {
    return (
      <div className="max-w-md mx-auto p-6 bg-white border border-gray-200 rounded-lg shadow-sm">
        <div className="text-center space-y-4">
          <h2 className="text-xl font-semibold">Sign in with Base</h2>
          <p className="text-gray-600">
            Connect your Base Account to get started
          </p>
          
          <button
            onClick={handleSignInWithBase}
            disabled={isConnecting}
            className="w-full px-4 py-2 bg-blue-500 text-white rounded-lg hover:bg-blue-600 disabled:opacity-50 disabled:cursor-not-allowed"
          >
            {isConnecting ? 'Connecting...' : 'Sign in with Base Account'}
          </button>
        </div>
      </div>
    )
  }

  return (
    <div className="max-w-md mx-auto p-6 bg-white border border-gray-200 rounded-lg shadow-sm">
      <div className="space-y-4">
        <div className="flex items-center justify-between">
          <h2 className="text-xl font-semibold">Connected</h2>
          <div className="w-3 h-3 bg-green-500 rounded-full"></div>
        </div>

        <div className="space-y-2">
          <p className="text-sm text-gray-600">User ID: {user?.id}</p>
          {user?.email && (
            <p className="text-sm text-gray-600">Email: {user.email.address}</p>
          )}
          
          {baseAccountWallet && (
            <div className="p-3 bg-blue-50 border border-blue-200 rounded">
              <p className="text-sm font-medium">Base Account</p>
              <p className="text-xs text-gray-600 font-mono">
                {baseAccountWallet.address}
              </p>
            </div>
          )}
        </div>

        <button
          onClick={logout}
          className="w-full px-4 py-2 bg-red-500 text-white rounded-lg hover:bg-red-600"
        >
          Logout
        </button>
      </div>
    </div>
  )
}
```

### 2. Wallet Connection Management

Handle Base Account connection and disconnection:

```tsx
// components/WalletConnection.tsx
import { usePrivy } from '@privy-io/react-auth'
import { useWallets } from '@privy-io/react-auth'
import { useState } from 'react'

export function WalletConnection() {
  const { authenticated, linkWallet, unlinkWallet } = usePrivy()
  const { wallets } = useWallets()
  const [isLinking, setIsLinking] = useState(false)

  const baseAccountWallet = wallets.find(
    wallet => wallet.walletClientType === 'base_account'
  )

  const embeddedWallet = wallets.find(
    wallet => wallet.walletClientType === 'privy'
  )

  const handleConnectBaseAccount = async () => {
    setIsLinking(true)
    try {
      await linkWallet()
    } catch (error) {
      console.error('Failed to link wallet:', error)
    } finally {
      setIsLinking(false)
    }
  }

  const handleDisconnectBaseAccount = async () => {
    if (baseAccountWallet) {
      try {
        await unlinkWallet(baseAccountWallet.address)
      } catch (error) {
        console.error('Failed to unlink wallet:', error)
      }
    }
  }

  if (!authenticated) {
    return <p>Please authenticate first</p>
  }

  return (
    <div className="space-y-4">
      <h3 className="text-lg font-semibold">Wallet Management</h3>
      
      {embeddedWallet && (
        <div className="p-4 bg-green-50 border border-green-200 rounded-lg">
          <div className="flex items-center justify-between">
            <div>
              <h4 className="font-medium">Embedded Wallet</h4>
              <p className="text-sm text-gray-600 font-mono">
                {embeddedWallet.address}
              </p>
            </div>
            <div className="w-2 h-2 bg-green-500 rounded-full"></div>
          </div>
        </div>
      )}

      {baseAccountWallet ? (
        <div className="p-4 bg-blue-50 border border-blue-200 rounded-lg">
          <div className="flex items-center justify-between">
            <div>
              <h4 className="font-medium">Base Account</h4>
              <p className="text-sm text-gray-600 font-mono">
                {baseAccountWallet.address}
              </p>
            </div>
            <button
              onClick={handleDisconnectBaseAccount}
              className="px-3 py-1 text-sm bg-red-500 text-white rounded hover:bg-red-600"
            >
              Disconnect
            </button>
          </div>
        </div>
      ) : (
        <div className="p-4 border-2 border-dashed border-gray-300 rounded-lg">
          <div className="text-center space-y-2">
            <p className="text-gray-600">No Base Account connected</p>
            <button
              onClick={handleConnectBaseAccount}
              disabled={isLinking}
              className="px-4 py-2 bg-blue-500 text-white rounded hover:bg-blue-600 disabled:opacity-50"
            >
              {isLinking ? 'Connecting...' : 'Connect Base Account'}
            </button>
          </div>
        </div>
      )}

      <div className="text-sm text-gray-600">
        Total connected wallets: {wallets.length}
      </div>
    </div>
  )
}
```

### 3. Transaction Handling

Use connected wallets for transactions:

```tsx
// components/TransactionManager.tsx
import { usePrivy } from '@privy-io/react-auth'
import { useWallets } from '@privy-io/react-auth'
import { useState } from 'react'

export function TransactionManager() {
  const { authenticated } = usePrivy()
  const { wallets } = useWallets()
  const [recipient, setRecipient] = useState('')
  const [amount, setAmount] = useState('')
  const [isLoading, setIsLoading] = useState(false)

  const baseAccountWallet = wallets.find(
    wallet => wallet.walletClientType === 'base_account'
  )

  const sendTransaction = async () => {
    if (!baseAccountWallet || !recipient || !amount) return

    setIsLoading(true)
    try {
      // Switch to Base network if needed
      await baseAccountWallet.switchChain(8453) // Base mainnet

      // Send transaction
      const txHash = await baseAccountWallet.sendTransaction({
        to: recipient as `0x${string}`,
        value: amount,
      })

      console.log('Transaction sent:', txHash)
      alert(`Transaction sent: ${txHash}`)
    } catch (error) {
      console.error('Transaction failed:', error)
      alert('Transaction failed')
    } finally {
      setIsLoading(false)
    }
  }

  if (!authenticated) {
    return <p>Please authenticate first</p>
  }

  if (!baseAccountWallet) {
    return <p>Please connect your Base Account first</p>
  }

  return (
    <div className="space-y-4">
      <h3 className="text-lg font-semibold">Send Transaction</h3>
      
      <div>
        <label className="block mb-2 text-sm font-medium">Recipient:</label>
        <input
          type="text"
          value={recipient}
          onChange={(e) => setRecipient(e.target.value)}
          className="w-full p-2 border border-gray-300 rounded-lg"
          placeholder="0x..."
        />
      </div>

      <div>
        <label className="block mb-2 text-sm font-medium">Amount (ETH):</label>
        <input
          type="number"
          value={amount}
          onChange={(e) => setAmount(e.target.value)}
          className="w-full p-2 border border-gray-300 rounded-lg"
          placeholder="0.01"
          step="0.001"
        />
      </div>

      <button
        onClick={sendTransaction}
        disabled={!recipient || !amount || isLoading}
        className="w-full px-4 py-2 bg-green-500 text-white rounded-lg hover:bg-green-600 disabled:opacity-50 disabled:cursor-not-allowed"
      >
        {isLoading ? 'Sending...' : 'Send Transaction'}
      </button>
    </div>
  )
}
```

### 4. Complete Authentication Flow

Here's a complete example that combines all components:

```tsx
// components/BaseAccountApp.tsx
import { BaseAccountAuth } from './BaseAccountAuth'
import { WalletConnection } from './WalletConnection'
import { TransactionManager } from './TransactionManager'
import { usePrivy } from '@privy-io/react-auth'

export function BaseAccountApp() {
  const { authenticated } = usePrivy()

  return (
    <div className="max-w-4xl mx-auto p-8 space-y-8">
      <h1 className="text-3xl font-bold text-center">
        Base Account with Privy
      </h1>

      <BaseAccountAuth />

      {authenticated && (
        <div className="grid md:grid-cols-2 gap-8">
          <WalletConnection />
          <TransactionManager />
        </div>
      )}
    </div>
  )
}
```

## Advanced Features

### Custom Login UI

You can customize the login experience:

```tsx
import { useLogin } from '@privy-io/react-auth'

export function CustomLogin() {
  const { login } = useLogin({
    onComplete: (user, isNewUser) => {
      console.log('Login completed:', { user, isNewUser })
    },
    onError: (error) => {
      console.error('Login error:', error)
    }
  })

  return (
    <div className="space-y-4">
      <button
        onClick={() => login({ loginMethod: 'wallet' })}
        className="w-full px-4 py-2 bg-blue-500 text-white rounded"
      >
        Connect Wallet
      </button>
      
      <button
        onClick={() => login({ loginMethod: 'email' })}
        className="w-full px-4 py-2 bg-green-500 text-white rounded"
      >
        Login with Email
      </button>
    </div>
  )
}
```

### Error Handling

Implement comprehensive error handling:

```tsx
import { usePrivy } from '@privy-io/react-auth'
import { useState, useEffect } from 'react'

export function ErrorHandler() {
  const { authenticated } = usePrivy()
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    const handleError = (event: any) => {
      setError(event.detail?.message || 'An error occurred')
    }

    window.addEventListener('privy:error', handleError)
    return () => window.removeEventListener('privy:error', handleError)
  }, [])

  if (error) {
    return (
      <div className="p-4 bg-red-50 border border-red-200 rounded-lg">
        <p className="text-red-700">{error}</p>
        <button
          onClick={() => setError(null)}
          className="mt-2 px-3 py-1 bg-red-500 text-white rounded text-sm"
        >
          Dismiss
        </button>
      </div>
    )
  }

  return null
}
```
