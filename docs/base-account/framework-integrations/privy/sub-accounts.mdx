Learn how to create and manage Base Account Sub Accounts using Privy's embedded wallets to improve user experience and reduce signature friction.

## Prerequisites

Make sure you have:
- [Set up Privy with Base Account](/base-account/framework-integrations/privy/setup)
- [Implemented Sign in with Base](/base-account/framework-integrations/privy/sign-in-with-base)

## Overview

Sub Accounts are app-specific wallets with their own addresses and signing capabilities that can transact using the parent account's balance through [Spend Permissions](/base-account/improve-ux/spend-permissions). This significantly improves user experience for high-volume signature scenarios.

Key benefits:
- **Reduced friction**: No passkey prompts for each transaction
- **App-specific**: Each Sub Account is unique to your application
- **Parent account balance**: Can spend from the main wallet using Spend Permissions
- **Seamless UX**: Users don't need to constantly approve transactions

## Creating Sub Accounts

### 1. Set Up the Sub Account Hook

Create a custom hook to manage Sub Account operations:

```tsx
// hooks/useSubAccounts.ts
import { usePrivy } from '@privy-io/react-auth'
import { useWallets } from '@privy-io/react-auth'
import { useState } from 'react'

export function useSubAccounts() {
  const { authenticated } = usePrivy()
  const { wallets } = useWallets()
  const [isLoading, setIsLoading] = useState(false)

  const baseAccountWallet = wallets.find(
    wallet => wallet.walletClientType === 'base_account'
  )

  const embeddedWallet = wallets.find(
    wallet => wallet.walletClientType === 'privy'
  )

  const createSubAccount = async () => {
    if (!baseAccountWallet || !embeddedWallet) {
      throw new Error('Base Account and embedded wallet required')
    }

    setIsLoading(true)
    try {
      // Switch to Base network
      await baseAccountWallet.switchChain(8453)

      // Check for existing Sub Account first
      const existingSubAccounts = await baseAccountWallet.request({
        method: 'wallet_getSubAccounts',
        params: [{
          version: '1.0',
          chainId: '0x2105', // Base mainnet
          from: baseAccountWallet.address,
        }]
      })

      // Create new Sub Account if none exists
      if (!existingSubAccounts?.subAccounts?.length) {
        const result = await baseAccountWallet.request({
          method: 'wallet_addSubAccount',
          params: [{
            version: '1.0',
            chainId: '0x2105',
            from: baseAccountWallet.address,
          }]
        })

        return result?.subAccount
      }

      return existingSubAccounts.subAccounts[0]
    } catch (error) {
      console.error('Failed to create Sub Account:', error)
      throw error
    } finally {
      setIsLoading(false)
    }
  }

  const getSubAccounts = async () => {
    if (!baseAccountWallet) return []

    try {
      const result = await baseAccountWallet.request({
        method: 'wallet_getSubAccounts',
        params: [{
          version: '1.0',
          chainId: '0x2105',
          from: baseAccountWallet.address,
        }]
      })

      return result?.subAccounts || []
    } catch (error) {
      console.error('Failed to get Sub Accounts:', error)
      return []
    }
  }

  return {
    authenticated,
    baseAccountWallet,
    embeddedWallet,
    createSubAccount,
    getSubAccounts,
    isLoading,
  }
}
```

### 2. Sub Account Manager Component

Create a component to manage Sub Account creation and display:

```tsx
// components/SubAccountManager.tsx
import { useSubAccounts } from '../hooks/useSubAccounts'
import { useState, useEffect } from 'react'

export function SubAccountManager() {
  const { 
    authenticated, 
    baseAccountWallet, 
    embeddedWallet, 
    createSubAccount, 
    getSubAccounts, 
    isLoading 
  } = useSubAccounts()
  
  const [subAccounts, setSubAccounts] = useState<string[]>([])
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    if (authenticated && baseAccountWallet) {
      loadSubAccounts()
    }
  }, [authenticated, baseAccountWallet])

  const loadSubAccounts = async () => {
    try {
      const accounts = await getSubAccounts()
      setSubAccounts(accounts)
    } catch (err) {
      setError('Failed to load Sub Accounts')
    }
  }

  const handleCreateSubAccount = async () => {
    try {
      setError(null)
      const newSubAccount = await createSubAccount()
      if (newSubAccount) {
        await loadSubAccounts() // Refresh the list
      }
    } catch (err) {
      setError('Failed to create Sub Account')
    }
  }

  if (!authenticated) {
    return (
      <div className="p-4 bg-yellow-50 border border-yellow-200 rounded-lg">
        <p className="text-yellow-700">Please log in to manage Sub Accounts</p>
      </div>
    )
  }

  if (!baseAccountWallet || !embeddedWallet) {
    return (
      <div className="p-4 bg-red-50 border border-red-200 rounded-lg">
        <p className="text-red-700">Both Base Account and embedded wallet are required for Sub Accounts</p>
      </div>
    )
  }

  return (
    <div className="space-y-6">
      <div className="flex items-center justify-between">
        <h2 className="text-xl font-semibold">Sub Account Manager</h2>
        <button
          onClick={handleCreateSubAccount}
          disabled={isLoading}
          className="px-4 py-2 bg-blue-500 text-white rounded-lg hover:bg-blue-600 disabled:opacity-50"
        >
          {isLoading ? 'Creating...' : 'Create Sub Account'}
        </button>
      </div>

      {error && (
        <div className="p-4 bg-red-50 border border-red-200 rounded-lg">
          <p className="text-red-700">{error}</p>
        </div>
      )}

      <div className="space-y-4">
        <h3 className="text-lg font-medium">Your Sub Accounts ({subAccounts.length})</h3>
        
        {subAccounts.length === 0 ? (
          <div className="p-6 bg-gray-50 border-2 border-dashed border-gray-300 rounded-lg text-center">
            <p className="text-gray-600">No Sub Accounts found</p>
            <p className="text-sm text-gray-500 mt-1">Create one to get started!</p>
          </div>
        ) : (
          <div className="space-y-3">
            {subAccounts.map((subAccount, index) => (
              <div
                key={subAccount}
                className="p-4 bg-blue-50 border border-blue-200 rounded-lg"
              >
                <div className="flex items-center justify-between">
                  <div>
                    <p className="font-mono text-sm">{subAccount}</p>
                    <p className="text-xs text-blue-600">Sub Account #{index + 1}</p>
                  </div>
                  <div className="w-3 h-3 bg-green-500 rounded-full" title="Active"></div>
                </div>
              </div>
            ))}
          </div>
        )}
      </div>

      <div className="p-4 bg-blue-50 border border-blue-200 rounded-lg">
        <h4 className="font-medium mb-2">Parent Account</h4>
        <p className="font-mono text-sm">{baseAccountWallet.address}</p>
      </div>
    </div>
  )
}
```

### 3. Sub Account Transaction Handler

Create a component to handle transactions using Sub Accounts:

```tsx
// components/SubAccountTransactions.tsx
import { useSubAccounts } from '../hooks/useSubAccounts'
import { useState, useEffect } from 'react'

export function SubAccountTransactions() {
  const { 
    authenticated, 
    embeddedWallet, 
    getSubAccounts 
  } = useSubAccounts()
  
  const [subAccounts, setSubAccounts] = useState<string[]>([])
  const [selectedSubAccount, setSelectedSubAccount] = useState('')
  const [recipient, setRecipient] = useState('')
  const [amount, setAmount] = useState('')
  const [isLoading, setIsLoading] = useState(false)

  useEffect(() => {
    if (authenticated) {
      loadSubAccounts()
    }
  }, [authenticated])

  const loadSubAccounts = async () => {
    const accounts = await getSubAccounts()
    setSubAccounts(accounts)
    if (accounts.length > 0 && !selectedSubAccount) {
      setSelectedSubAccount(accounts[0])
    }
  }

  const sendTransactionFromSubAccount = async () => {
    if (!embeddedWallet || !selectedSubAccount || !recipient || !amount) return

    setIsLoading(true)
    try {
      // Use embedded wallet to control the Sub Account
      // This avoids passkey prompts since embedded wallet handles signing
      const txHash = await embeddedWallet.sendTransaction({
        to: recipient as `0x${string}`,
        value: amount,
        // The transaction will be sent from the Sub Account
        // but signed by the embedded wallet
      })

      console.log('Transaction sent from Sub Account:', txHash)
      alert(`Transaction successful: ${txHash}`)
      
      // Reset form
      setRecipient('')
      setAmount('')
    } catch (error) {
      console.error('Sub Account transaction failed:', error)
      alert('Transaction failed')
    } finally {
      setIsLoading(false)
    }
  }

  if (!authenticated || subAccounts.length === 0) {
    return (
      <div className="p-4 bg-yellow-50 border border-yellow-200 rounded-lg">
        <p className="text-yellow-700">
          Please create a Sub Account first to send transactions
        </p>
      </div>
    )
  }

  return (
    <div className="space-y-4">
      <h3 className="text-lg font-semibold">Send from Sub Account</h3>
      
      <div>
        <label className="block mb-2 text-sm font-medium">Sub Account:</label>
        <select
          value={selectedSubAccount}
          onChange={(e) => setSelectedSubAccount(e.target.value)}
          className="w-full p-2 border border-gray-300 rounded-lg"
        >
          {subAccounts.map((account, index) => (
            <option key={account} value={account}>
              Sub Account #{index + 1}: {account.slice(0, 10)}...
            </option>
          ))}
        </select>
      </div>

      <div>
        <label className="block mb-2 text-sm font-medium">Recipient:</label>
        <input
          type="text"
          value={recipient}
          onChange={(e) => setRecipient(e.target.value)}
          className="w-full p-2 border border-gray-300 rounded-lg"
          placeholder="0x..."
        />
      </div>

      <div>
        <label className="block mb-2 text-sm font-medium">Amount (ETH):</label>
        <input
          type="number"
          value={amount}
          onChange={(e) => setAmount(e.target.value)}
          className="w-full p-2 border border-gray-300 rounded-lg"
          placeholder="0.01"
          step="0.001"
        />
      </div>

      <button
        onClick={sendTransactionFromSubAccount}
        disabled={!selectedSubAccount || !recipient || !amount || isLoading}
        className="w-full px-4 py-2 bg-green-500 text-white rounded-lg hover:bg-green-600 disabled:opacity-50"
      >
        {isLoading ? 'Sending...' : 'Send from Sub Account'}
      </button>

      <div className="text-sm text-gray-600">
        <p>💡 <strong>Note:</strong> This transaction will be sent from your Sub Account using your embedded wallet for signing, avoiding passkey prompts.</p>
      </div>
    </div>
  )
}
```

## Advanced Configuration

### Setting Up Spend Permissions

Configure Spend Permissions to allow Sub Accounts to spend from the parent account:

```tsx
// utils/spendPermissions.ts
export const createSpendPermission = async (
  baseAccountWallet: any,
  subAccountAddress: string,
  spendLimit: string
) => {
  try {
    const result = await baseAccountWallet.request({
      method: 'coinbase_fetchPermissions',
      params: [{
        account: subAccountAddress,
        spender: subAccountAddress,
        token: '0x0000000000000000000000000000000000000000', // ETH
        allowance: spendLimit,
      }]
    })

    return result
  } catch (error) {
    console.error('Failed to create spend permission:', error)
    throw error
  }
}
```

### Error Handling

Implement comprehensive error handling:

```tsx
export function SubAccountErrorBoundary({ children }: { children: React.ReactNode }) {
  const [error, setError] = useState<string | null>(null)

  const handleError = (error: Error) => {
    console.error('Sub Account error:', error)
    
    if (error.message.includes('User rejected')) {
      setError('Transaction was cancelled by user')
    } else if (error.message.includes('insufficient funds')) {
      setError('Insufficient funds for this transaction')
    } else {
      setError('An unexpected error occurred')
    }
  }

  if (error) {
    return (
      <div className="p-4 bg-red-50 border border-red-200 rounded-lg">
        <p className="text-red-700">{error}</p>
        <button
          onClick={() => setError(null)}
          className="mt-2 px-3 py-1 bg-red-500 text-white rounded text-sm"
        >
          Dismiss
        </button>
      </div>
    )
  }

  return <>{children}</>
}
```
